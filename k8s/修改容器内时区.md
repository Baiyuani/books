


# 配置容器内时区

```yaml
# 变量传递TZ
      containers:
      - env:
        - name: TZ
          value: Asia/Shanghai
---     
# 将宿主机的/etc/localtime挂载进容器
        volumeMounts:
        - mountPath: /etc/localtime
          name: localtime
      volumes:
      - hostPath:
          path: /etc/localtime
          type: ""
        name: localtime
---
# 如果挂载/etc/localtime没有作用：
        lifecycle:
          postStart:
            exec:
              command:
              - /bin/sh
              - -c
              - ln -s /usr/share/zoneinfo/Asia/Shanghai /etc/localtime
              
        volumeMounts:
        - mountPath: /usr/share/zoneinfo/Asia/Shanghai
          name: localtime
      volumes:
      - hostPath:
          path: /usr/share/zoneinfo/Asia/Shanghai
          type: ""
        name: localtime
```
