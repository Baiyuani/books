# centos6

```shell
vim /etc/ntp.conf

# 立即找服务器同步时间，危险操作！时间差异较大时直接同步可能会出问题
ntpd -gq

service ntpd start 
# 开机自启
chkconfig ntpd on 
```

# centos7

```shell
vim /etc/chronyd.conf

#查看时间同步源：
chronyc sources -v

systemctl start chronyd

#立即手工同步,危险操作！时间差异较大时直接同步可能会出问题
chronyc -a makestep

systemctl enable chronyd
```
